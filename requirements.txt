# Cloudflare bypass — TLS-level fingerprinting + JS challenge solving.
# Both are needed because Cloudflare runs MULTIPLE detection layers:
#   1. TLS/JA3 fingerprint check (defeated by curl_cffi's real-Chrome handshake)
#   2. JavaScript challenge (defeated by cloudscraper's interpreter)
#   3. Turnstile / v3 VM challenges (only the enhanced cloudscraper handles these)
# Bumped curl_cffi to a version with current Chrome impersonation profiles.
curl_cffi>=0.7.0
# Enhanced cloudscraper from VeNoMouS GitHub fork — supports Cloudflare v2,
# v3 (JavaScript VM), and Turnstile challenges that the old PyPI version
# (stuck at 1.2.71) cannot solve. fp.net specifically requires this fork
# to scrape reliably. If the GitHub install fails for any reason, the bot
# falls back to plain requests/curl_cffi.
cloudscraper @ git+https://github.com/VeNoMouS/cloudscraper.git@3.0.0
# Polymarket CLOB client — Polymarket reads/writes.
# Pinned to a known-good version that matches the import paths the code uses
# (py_clob_client.client.ClobClient, py_clob_client.clob_types.ApiCreds).
py-clob-client==0.34.6
# web3 + eth-* — required by the INLINED Limitless SDK (search for "LIMITLESS
# EXCHANGE SDK" in app.py). The SDK code itself is bundled in the same file;
# only the pip packages it imports need to be listed here.
#
# py-clob-client 0.34.6 requires eth-account>=0.13.0. web3 7.x is the first
# web3 line that loosened its eth-account upper bound to allow 0.13.x, so
# pinning web3>=7 makes the resolver pick exactly one eth-account version
# (the latest 0.13.x) and avoids the eth-account==0.10.0 conflict that
# py-limitless used to drag in. The SDK code uses only the snake_case web3
# API (to_checksum_address, get_transaction_count, send_raw_transaction,
# build_transaction) so it runs on web3 v7 unmodified.
web3>=7.0
